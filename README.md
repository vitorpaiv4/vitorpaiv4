int main() {
    float nota1, nota2, media1, media2;

    printf("Digite a primeira nota: ");

    printf("Digite a segunda nota: ");
    scanf("%f", &nota2);

    media1 = (nota1 + nota1) / 2;
    media2 = (nota2 + nota2) / 2;

    printf("\nMédia da primeira nota: %.2f\n", media1);

    return 0;
}
